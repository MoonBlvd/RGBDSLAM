FIND_PACKAGE( PCL 1.2 REQUIRED COMPONENTS common io visualization)
FIND_PACKAGE( OpenCV REQUIRED )

ADD_DEFINITIONS( ${PCL_DEFINITIONS} )
INCLUDE_DIRECTORIES( ${PCL_INCLUDE_DIRS}  )
LINK_LIBRARIES( ${PCL_LIBRARY_DIRS} )

ADD_LIBRARY(PointCloud PointCloud.cpp)
TARGET_LINK_LIBRARIES(PointCloud ${OpenCV_LIBS} ${PCL_LIBRARIES} )

ADD_EXECUTABLE(main main.cpp )
TARGET_LINK_LIBRARIES( main PointCloud ${OpenCV_LIBS} ${PCL_LIBRARIES} )
